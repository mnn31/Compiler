PROCEDURE add3();
count := count + 3;

PROCEDURE countUp(count, max);
IF count <= max THEN
BEGIN
   WRITELN(count);
   ignore := countUp(count + 1, max);
END;

PROCEDURE foo(d);
ignore := bar(d + f);

PROCEDURE bar(f);
WRITELN(f);

PROCEDURE print(n);
WRITELN(n);

BEGIN
   count := 2;
   ignore := add3();
   WRITELN(count);

   ignore := countUp(1, 5);

   f := 2;
   ignore := foo(3);
   WRITELN(f);

   n := 3;
   ignore := print(5);
   WRITELN(n);
END;
.
