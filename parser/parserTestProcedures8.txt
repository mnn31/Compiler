// Multiple procedures calling each other with return values
// Tests nested calls: isEven calls isOdd and vice versa (mutual recursion)
// Expected output: 1 0 1 0 1 (isEven for 0,1,2,3,4)
PROCEDURE isEven(n);
BEGIN
   IF n = 0 THEN
   BEGIN
      isEven := 1;
   END;
   ELSE
   BEGIN
      isEven := isOdd(n - 1);
   END;
END;
PROCEDURE isOdd(n);
BEGIN
   IF n = 0 THEN
   BEGIN
      isOdd := 0;
   END;
   ELSE
   BEGIN
      isOdd := isEven(n - 1);
   END;
END;
BEGIN
   WRITELN(isEven(0));
   WRITELN(isEven(1));
   WRITELN(isEven(2));
   WRITELN(isEven(3));
   WRITELN(isEven(4));
END;
.
